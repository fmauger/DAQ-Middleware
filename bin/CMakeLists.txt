
configure_file(condition_xml2json.in ${CMAKE_CURRENT_BINARY_DIR}/condition_xml2json @ONLY)
configure_file(run.py.in ${CMAKE_CURRENT_BINARY_DIR}/run.py @ONLY)
configure_file(run.py3.in ${CMAKE_CURRENT_BINARY_DIR}/run.py3 @ONLY)

set(DAQMW_BIN_FILES
  ${CMAKE_CURRENT_BINARY_DIR}/condition_xml2json
  condition_xml2json.py
  xml2json-with-attribute.xslt
  ${CMAKE_CURRENT_BINARY_DIR}/run.py
  ${CMAKE_CURRENT_BINARY_DIR}/run.py3
  )

file(COPY ${DAQMW_BIN_FILES}
  DESTINATION "${CMAKE_RUNTIME_OUTPUT_DIRECTORY}"
  FILE_PERMISSIONS
  OWNER_READ OWNER_WRITE OWNER_EXECUTE
  GROUP_READ GROUP_EXECUTE
  WORLD_READ WORLD_EXECUTE
  )

install(FILES ${DAQMW_BIN_FILES}
  DESTINATION "${CMAKE_INSTALL_BINDIR}"
  PERMISSIONS OWNER_READ OWNER_WRITE OWNER_EXECUTE
  GROUP_READ GROUP_EXECUTE
  WORLD_READ WORLD_EXECUTE
  )
